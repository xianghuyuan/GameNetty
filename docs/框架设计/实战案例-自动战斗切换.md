# 实战案例：实现自动战斗切换功能

## 完整开发流程演示

### 步骤 1：定义协议（已存在）

// ResponseType M2C_SwitchControlMode
message C2M_SwitchControlMode // ISessionRequest
{
    int32 RpcId = 1;
    int32 Mode = 2; // 0: Auto, 1: Manual
}

message M2C_SwitchControlMode // ISessionResponse
{
    int32 RpcId = 1;
    int32 Error = 2;

{
    int64 UnitId = 1;

### 步骤 2：服务端实现

#### 2.1 定义组件（Model）

```csharp
// Server/Model/Demo/Battle/PlayerCombatModeComponent.cs

namespace ET.Server
{
    public enum BattleMode
    {
        Manual = 0,   // 手动模式
        Auto = 1,     // 自动模式
    }
    
    [ComponentOf(typeof(BattleUnit))]
    public class PlayerCombatModeComponent : Entity, IAwake, IDestroy
    {
        public BattleMode Mode { get; set; } = BattleMode.Manual;
        public EntityRef<SimpleAIComponent> AutoAI { get; set; }
        public bool IsAutoBattle => Mode == BattleMode.Auto;
    }
}
```

#### 2.2 实现系统逻辑（Hotfix）

```csharp
// Server/Hotfix/Demo/Battle/PlayerCombatModeComponentSystem.cs

namespace ET.Server
{
    [EntitySystemOf(typeof(PlayerCombatModeComponent))]
    public static partial class PlayerCombatModeComponentSystem
    {
        [EntitySystem]
        private static void Awake(this PlayerCombatModeComponent self)
        {
            self.Mode = BattleMode.Manual;
        }
        
        public static void SetMode(this PlayerCombatModeComponent self, BattleMode mode)
        {
            if (self.Mode == mode) return;
            
            self.Mode = mode;
            BattleUnit owner = self.GetParent<BattleUnit>();
            
            if (mode == BattleMode.Auto)
            {
                // 创建 AI 组件
                SimpleAIComponent ai = owner.AddComponent<SimpleAIComponent>();
                self.AutoAI = ai;
            }
            else
            {
                // 移除 AI 组件
                SimpleAIComponent ai = self.AutoAI;
                ai?.Dispose();
            }
        }
    }
}
```

#### 2.3 实现消息处理器

```csharp
// Server/Hotfix/Demo/Battle/Handler/C2M_SwitchControlModeHandler.cs

namespace ET.Server
{
    [MessageHandler(SceneType.Map)]
    public class C2M_SwitchControlModeHandler : MessageHandler<Scene, C2M_SwitchControlMode, M2C_SwitchControlMode>
    {
        protected override async ETTask Run(Scene scene, C2M_SwitchControlMode request, M2C_SwitchControlMode response)
        {
            // 1. 查找玩家战斗单位
            BattleUnit playerUnit = FindPlayerBattleUnit(scene);
            if (playerUnit == null)
            {
                response.Error = ErrorCode.ERR_BattleUnitNotFound;
                return;
            }
            
            // 2. 获取或添加模式组件
            PlayerCombatModeComponent modeComp = playerUnit.GetComponent<PlayerCombatModeComponent>();
            if (modeComp == null)
            {
                modeComp = playerUnit.AddComponent<PlayerCombatModeComponent>();
            }
            
            // 3. 切换模式
            BattleMode newMode = (BattleMode)request.Mode;
            modeComp.SetMode(newMode);
            
            // 4. 返回结果
            response.CurrentMode = (int)modeComp.Mode;
            
            // 5. 广播给其他玩家
            BattleRoom battleRoom = playerUnit.GetParent<BattleRoom>();
            M2C_ControlModeChanged broadcastMsg = M2C_ControlModeChanged.Create();
            broadcastMsg.UnitId = playerUnit.Id;
            broadcastMsg.NewMode = (int)modeComp.Mode;
            battleRoom.BroadcastToPlayers(broadcastMsg);
            
            await ETTask.CompletedTask;
        }
    }
}
```

```csharp
// Unity/Assets/GameScripts/HotFix/GameLogic/Module/Battle/BattleModeChanged_UpdateUI.cs

namespace ET
{
    [Event(SceneType.Main)]
    public class BattleModeChanged_UpdateUI : AEvent<Scene, BattleModeChanged>
    {
        protected override async ETTask Run(Scene scene, BattleModeChanged args)
        {
            // 更新 UI 显示
            var battleUI = await GameModule.UI.GetUIAsyncAwait<GameLogic.BattleMainUI>();
            if (battleUI != null)
            {
                battleUI.UpdateModeButton(args.NewMode);
            }
            
            // 显示提示
            string modeText = args.NewMode == 0 ? "手动模式" : "自动模式";
            GameModule.UI.ShowToast($"切换到{modeText}");
            
            await ETTask.CompletedTask;
        }
    }
}
```

### 步骤 5：客户端消息监听

// Unity/Assets/GameScripts/HotFix/GameLogic/Module/Battle/M2C_ControlModeChangedHandler.cs

namespace ET
{
    [MessageHandler(SceneType.Main)]
    public class M2C_ControlModeChangedHandler : MessageHandler<Scene, M2C_ControlModeChanged>
    {
        protected override async ETTask Run(Scene scene, M2C_ControlModeChanged message)
        {
            // 发布事件
            EventSystem.Instance.Publish(scene, new BattleModeChanged
            {
                UnitId = message.UnitId,
                NewMode = message.NewMode
            });
            
            await ETTask.CompletedTask;
        }
    }
}

### 步骤 6：TEngine UI 实现

```csharp
// Unity/Assets/GameScripts/HotFix/GameLogic/UI/BattleMainUI/BattleMainUI.cs

namespace GameLogic
{
    [Window(UILayer.UI, false)]
    public class BattleMainUI : UIWindow
    {
        private Button btnAutoMode;
        private Text txtModeStatus;
        private int currentMode = 0; // 0=Manual, 1=Auto
        
        protected override void OnCreate()
        {
            // 绑定按钮
            btnAutoMode = GetWidget<Button>("BtnAutoMode");
            txtModeStatus = GetWidget<Text>("TxtModeStatus");
            
            btnAutoMode.onClick.AddListener(OnAutoModeButtonClick);
        }
        
        private async void OnAutoModeButtonClick()
        {
            // 切换模式
            int newMode = currentMode == 0 ? 1 : 0;
            
            // 调用 ET 逻辑层
            Scene clientScene = Client.Instance.CurrentScene();
            Session session = clientScene.GetComponent<SessionComponent>().Session;
            
            C2M_SwitchControlMode request = C2M_SwitchControlMode.Create();
            request.Mode = newMode;
            
            M2C_SwitchControlMode response = await session.Call(request) as M2C_SwitchControlMode;
            
            if (response.Error != ErrorCode.ERR_Success)
            {
                GameModule.UI.ShowToast($"切换失败: {response.Message}");
                return;
            }
            
            // 更新本地状态
            currentMode = response.CurrentMode;
            UpdateModeButton(currentMode);
        }
        
        public void UpdateModeButton(int mode)
        {
            currentMode = mode;
            
            if (mode == 0)
            {
                txtModeStatus.text = "手动";
                btnAutoMode.GetComponentInChildren<Text>().text = "切换自动";
            }
            else
            {
                txtModeStatus.text = "自动";
                btnAutoMode.GetComponentInChildren<Text>().text = "切换手动";
            }
        }
    }
}
```

## 关键要点

### 1. 职责分离

- **TEngine UI**：只负责显示和用户交互
- **ET Client**：负责网络通信和事件分发
- **ET Server**：负责所有游戏逻辑和数据管理

### 2. 事件驱动

- 服务端广播 → 客户端消息处理器 → 发布事件 → 事件订阅者 → 更新 UI
- 解耦各个模块，避免直接依赖

### 3. 异步处理

- 所有网络请求都是异步的（async/await）
- 事件处理也是异步的（async ETTask）

### 4. 错误处理

- 每个环节都要检查错误码
- 向用户显示友好的错误提示

这就是 ET + TEngine 的完整开发流程！
