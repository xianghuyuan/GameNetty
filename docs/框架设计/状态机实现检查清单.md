```csharp
// Unity/Assets/Scripts/Hotfix/Server/Demo/Unit/StateFlags/IdleStateFlagSystem.cs
namespace ET.Server
{
    [EntitySystemOf(typeof(IdleStateFlag))]
    [FriendOf(typeof(IdleStateFlag))]
    public static partial class IdleStateFlagSystem
    {
        [EntitySystem]
        private static void Awake(this IdleStateFlag self)
        {
            self.EnterTime = TimeInfo.Instance.ServerNow();
            
            Unit unit = self.GetParent<Unit>();
            Log.Debug($"[State] Unit({unit.Id}) 进入待机状态");
        }

        [EntitySystem]
        private static void Destroy(this IdleStateFlag self)
        {
            Unit unit = self.GetParent<Unit>();
            long duration = TimeInfo.Instance.ServerNow() - self.EnterTime;
            Log.Debug($"[State] Unit({unit.Id}) 退出待机状态, 持续时长: {duration}ms");
        }
    }
}
```

### Step 3: 创建状态切换辅助类

```csharp
// Unity/Assets/Scripts/Hotfix/Server/Demo/Unit/StateHelper.cs
namespace ET.Server
{
    /// <summary>
    /// 状态切换辅助类（ET风格）
    /// </summary>
    public static class StateHelper
    {
        /// <summary>
        /// 切换到待机状态
        /// </summary>
        public static void ChangeToIdle(this Unit self)
        {
            // 检查是否已经在待机状态
            if (self.GetComponent<IdleStateFlag>() != null)
            {
                return;
            }
            
            // 检查防抖（从移动状态切换时）
            var moveFlag = self.GetComponent<MoveStateFlag>();
            if (moveFlag != null)
            {
                long now = TimeInfo.Instance.ServerNow();
                if (now - moveFlag.EnterTime < 100) // 100ms防抖
                {
                    Log.Debug($"[State] Unit({self.Id}) 防抖拒绝切换到Idle");
                    return;
                }
            }
            
            // 移除其他状态标记
            self.RemoveComponent<MoveStateFlag>();
            self.RemoveComponent<AttackStateFlag>();
            
            // 添加待机状态标记
            self.AddComponent<IdleStateFlag>();
        }
        
        /// <summary>
        /// 切换到移动状态
        /// </summary>
        public static void ChangeToMove(this Unit self)
        {
            // 检查是否已经在移动状态
            if (self.GetComponent<MoveStateFlag>() != null)
            {
                return;
            }
            
            // 检查防抖（从待机状态切换时）
            var idleFlag = self.GetComponent<IdleStateFlag>();
            if (idleFlag != null)
            {
                long now = TimeInfo.Instance.ServerNow();
                if (now - idleFlag.EnterTime < 200) // 200ms防抖
                {
                    Log.Debug($"[State] Unit({self.Id}) 防抖拒绝切换到Move");
                    return;
                }
            }
            
            // 移除其他状态标记
            self.RemoveComponent<IdleStateFlag>();
            self.RemoveComponent<AttackStateFlag>();
            
            // 添加移动状态标记
            self.AddComponent<MoveStateFlag>();
        }
        
        /// <summary>
        /// 切换到攻击状态
        /// </summary>
        public static void ChangeToAttack(this Unit self, long targetId)
        {
            // 检查是否已经在攻击状态
            var attackFlag = self.GetComponent<AttackStateFlag>();
            if (attackFlag != null)
            {
                // 更新目标ID
                attackFlag.TargetId = targetId;
                return;
            }
            
            // 移除其他状态标记
            self.RemoveComponent<IdleStateFlag>();
            self.RemoveComponent<MoveStateFlag>();
            
            // 添加攻击状态标记
            attackFlag = self.AddComponent<AttackStateFlag>();
            attackFlag.TargetId = targetId;
        }
        
        /// <summary>
        /// 检查是否在待机状态
        /// </summary>
        public static bool IsIdle(this Unit self)
        {
            return self.GetComponent<IdleStateFlag>() != null;
        }
        
        /// <summary>
        /// 检查是否在移动状态
        /// </summary>
        public static bool IsMoving(this Unit self)
        {
            return self.GetComponent<MoveStateFlag>() != null;
        }
        
        /// <summary>
        /// 检查是否在攻击状态
        /// </summary>
        public static bool IsAttacking(this Unit self)
        {
            return self.GetComponent<AttackStateFlag>() != null;
        }
        
        /// <summary>
        /// 获取当前状态名称（用于调试）
        /// </summary>
        public static string GetCurrentStateName(this Unit self)
        {
            if (self.IsIdle()) return "Idle";
            if (self.IsMoving()) return "Move";
            if (self.IsAttacking()) return "Attack";
            return "None";
        }
    }
}
```

```csharp
public static void EndCombat(this CombatComponent self)
{
    Unit unit = self.GetParent<Unit>();
    self.IsInCombat = false;
    self.TargetId = 0;
    
    // 切换到待机状态
```

#### 4.2 修改 PlayerAIComponentSystem.cs

```csharp
// 旧代码
public static void DecideAction(this PlayerAIComponent self)
{
    // ...
    if (dist > self.AttackRange)
    {
        unit.GetComponent<StateMachineComponent>().ChangeState(MachineState.Move);
        // ...
    }
}

// ✅ 新代码
public static void DecideAction(this PlayerAIComponent self)
{
    Unit unit = self.GetParent<Unit>();
    
    // 检查是否在攻击状态
    if (unit.IsAttacking())
    {
        // 正在攻击，不做处理
        return;
    }
    
    Unit target = AIHelper.FindNearestEnemy(unit, self.SearchRange);
    if (target == null) return;
    
    float dist = PositionHelper.Distance2D(unit, target);
    if (dist > self.AttackRange)
    {
        // 切换到移动状态
        unit.ChangeToMove();
        unit.FindPathMoveToAsync(target.Position).Coroutine();
    }
    else
    {
        // 切换到攻击状态
        unit.ChangeToAttack(target.Id);
        unit.GetComponent<CombatComponent>().Attack(target, true).Coroutine();
    }
}
```

```csharp
// ✅ 新代码
public static async ETTask<bool> MoveToAsync(this MoveComponent self, ...)
{
    Unit unit = self.GetParent<Unit>();
    
    // 切换到移动状态
    unit.ChangeToMove();
    
    // 执行移动逻辑...
    bool moveRet = await self.ExecuteMove(...);
    
    if (moveRet)
    {
        // 移动完成，切换到待机状态
        unit.ChangeToIdle();
    }
    
    return moveRet;
}
```
