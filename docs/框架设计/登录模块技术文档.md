## 1. 架构概述

### 1.1 整体架构

本登录模块基于 **ET框架** 的分布式服务器架构，采用多服务器节点协作完成登录流程：

| 服务器节点 | 职责 | 主要功能 |
|-----------|------|---------|
| **Realm服务器** | 账号验证服 | 账号密码验证、Token生成、Session管理 |
| **LoginCenter服务器** | 登录中心服 | 登录记录管理、顶号处理、多服登录协调 |
| **Gate服务器** | 网关服 | 客户端连接管理、Player实体创建、消息转发 |

### 1.2 登录流程概览

```
客户端 → Realm(账号验证) → LoginCenter(登录记录) → Gate(网关连接) → Map(进入游戏)
```

```csharp
namespace ET.Server
{
    [EntitySystemOf(typeof(TokenComponent))]
    [FriendOfAttribute(typeof(ET.Server.TokenComponent))]
    public static partial class TokenComponentSystem
    {
        [EntitySystem]
        private static void Awake(this ET.Server.TokenComponent self) { }

        /// <summary>
        /// 添加Token，自动设置10分钟超时
        /// </summary>
        public static void Add(this TokenComponent self, string key, string token)
        {
            self.TokenDictionary.Add(key, token);
            self.TimeOutRemoveKey(key, token).Coroutine();
        }

        public static string Get(this TokenComponent self, string key)
        {
            string value = string.Empty;
            self.TokenDictionary.TryGetValue(key, out value);
            return value;
        }

        public static void Remove(this TokenComponent self, string key)
        {
            if (self.TokenDictionary.ContainsKey(key))
            {
                self.TokenDictionary.Remove(key);
            }
        }

        /// <summary>
        /// Token超时自动移除（10分钟 = 600000ms）
        /// </summary>
        private static async ETTask TimeOutRemoveKey(this TokenComponent self, string key, string tokenkey)
        {
            await self.Root().GetComponent<TimerComponent>().WaitAsync(600000);
            string onlineToken = self.Get(key);
            if (!string.IsNullOrEmpty(onlineToken) && onlineToken == tokenkey)
            {
                self.Remove(key);
            }
        }
    }
}
```

## 4. 客户端实现

### 4.1 登录入口 (LoginHelper)

**文件位置**: `Unity/Assets/Scripts/Hotfix/Client/Demo/Main/Login/LoginHelper.cs`

```csharp
namespace ET.Client
{
    public static class LoginHelper
    {
        /// <summary>
        /// 登录主流程
        /// </summary>
        /// <param name="root">客户端Scene</param>
        /// <param name="account">账号</param>
        /// <param name="password">密码</param>
        public static async ETTask Login(Scene root, string account, string password)
        {
            // 1. 创建网络发送组件
            root.RemoveComponent<ClientSenderComponent>();
            ClientSenderComponent clientSenderComponent = root.AddComponent<ClientSenderComponent>();
            
            // 2. 发送登录请求到Realm
            NetClient2Main_Login response = await clientSenderComponent.LoginAsync(account, password);
            if (response.Error != ErrorCode.ERR_Success)
            {
                Log.Error($"请求登陆失败:{response.Error}");
                return;
            }
            Log.Debug("请求登录成功");
            string Token = response.Token;
            
            // 3. 获取服务器列表
            C2R_GetServerInfos c2RGetServerInfos = C2R_GetServerInfos.Create();
            c2RGetServerInfos.Account = account;
            c2RGetServerInfos.Token = response.Token;
            R2C_GetServerInfos r2CGetServerInfos = await clientSenderComponent.Call(c2RGetServerInfos) as R2C_GetServerInfos;
            if (r2CGetServerInfos.Error != ErrorCode.ERR_Success)
            {
                Log.Error("请求服务器列表失败");
                return;
            }

            ServerInfoProto serverInfoProto = r2CGetServerInfos.ServerInfosList[0];
            Log.Debug($"请求服务器列表成功，区服名称：{serverInfoProto.ServerName} 区服ID:{serverInfoProto.Id}");
            
            // 4. 获取区服角色列表
            C2R_GetRoles c2RGetRoles = C2R_GetRoles.Create();
            c2RGetRoles.Token = Token;
            c2RGetRoles.Account = account;
            c2RGetRoles.ServerId = serverInfoProto.Id;
            R2C_GetRoles r2CGetRoles = await clientSenderComponent.Call(c2RGetRoles) as R2C_GetRoles;
            if (r2CGetRoles.Error != ErrorCode.ERR_Success)
            {
                Log.Error("请求区服角色列表失败");
                return;
            }

            RoleInfoProto roleInfoProto = default;
            if (r2CGetRoles.RoleInfo.Count <= 0)
            {
                // 5. 没有角色信息，创建角色
                C2R_CreateRole c2RCreateRole = C2R_CreateRole.Create();
                c2RCreateRole.Token = Token;
                c2RCreateRole.Account = account;
                c2RCreateRole.ServerId = serverInfoProto.Id;
                c2RCreateRole.Name = account;

                R2C_CreateRole r2CCreateRole = await clientSenderComponent.Call(c2RCreateRole) as R2C_CreateRole;
                if (r2CCreateRole.Error != ErrorCode.ERR_Success)
                {
                    Log.Error("创建区服角色失败");
                    return;
                }
                roleInfoProto = r2CCreateRole.RoleInfo;
            }
            else
            {
                roleInfoProto = r2CGetRoles.RoleInfo[0];
            }

            // 6. 请求获取RealmKey（Gate网关连接凭证）
            C2R_GetRealmKey c2RGetRealmKey = C2R_GetRealmKey.Create();
            c2RGetRealmKey.Token = Token;
            c2RGetRealmKey.Account = account;
            c2RGetRealmKey.ServerId = serverInfoProto.Id;
            R2C_GetRealmKey r2CGetRealmKey = await clientSenderComponent.Call(c2RGetRealmKey) as R2C_GetRealmKey;

            if (r2CGetRealmKey.Error != ErrorCode.ERR_Success)
            {
                Log.Error("获取RealmKey失败");
                return;
            }
            
            // 7. 请求游戏角色进入Map地图
            NetClient2Main_LoginGame netClient2MainLoginGame =
                await clientSenderComponent.LoginGameAsync(account, r2CGetRealmKey.Key, roleInfoProto.Id, r2CGetRealmKey.Address);
            if (netClient2MainLoginGame.Error != ErrorCode.ERR_Success)
            {
                Log.Error($"进入游戏失败:{netClient2MainLoginGame.Error}");
                return;
            }

            // 8. 设置玩家ID
            root.Root().GetComponent<PlayerComponent>().MyId = netClient2MainLoginGame.PlayetId;
            Log.Debug("进入游戏成功");

            // 9. 发布登录完成事件
            await EventSystem.Instance.PublishAsync(root, new LoginFinish());
        }
    }
}
```

### 5.1 Realm服务器 - 账号登录Handler

**文件位置**: `Unity/Assets/Scripts/Hotfix/Server/Demo/Realm/Handler/C2R_LoginAccountHandler.cs`

```csharp
namespace ET.Server
{
    /// <summary>
    /// Realm公共服 - 账号登录处理
    /// </summary>
    [MessageSessionHandler(SceneType.Realm)]
    [FriendOfAttribute(typeof(Account))]
    public class C2R_LoginAccountHandler : MessageSessionHandler<C2R_LoginAccount, R2C_LoginAccount>
    {
        protected override async ETTask Run(Session session, C2R_LoginAccount request, R2C_LoginAccount response)
        {
            // 移除Session超时组件
            session.RemoveComponent<SessionAcceptTimeoutComponent>();

            // 1. 检查重复登录
            if (session.GetComponent<SessionLockingComponent>() != null)
            {
                response.Error = ErrorCode.ERR_RequestRepeatedly;
                session.Disconnect().Coroutine();
                return;
            }

            // 2. 验证账号密码非空
            if (string.IsNullOrEmpty(request.AccountName) || string.IsNullOrEmpty(request.Password))
            {
                response.Error = ErrorCode.ERR_LoginInfoIsNull;
                session.Disconnect().Coroutine();
                return;
            }

            // 3. 正则验证账号格式（6-15位，包含大小写字母和数字）
            if (!Regex.IsMatch(request.AccountName.Trim(), "^(?=.*[0-9].*)(?=.*[A-Z].*)(?=.*[a-z].*).{6,15}$"))
            {
                response.Error = ErrorCode.ERR_AccountNameFormError;
                session.Disconnect().Coroutine();
                return;
            }
            
            // 4. 验证账号只包含字母数字
            if (!Regex.IsMatch(request.AccountName.Trim(), "^[A-Za-z0-9]+$"))
            {
                response.Error = ErrorCode.ERR_PasswordFormError;
                session.Disconnect().Coroutine();
                return;
            }

            // 使用协程锁防止并发登录
            CoroutineLockComponent coroutineLockComponent = session.Root().GetComponent<CoroutineLockComponent>();
            using (session.AddComponent<SessionLockingComponent>())
            {
                using (await coroutineLockComponent.Wait(CoroutineLockType.LoginAccount, request.AccountName.GetLongHashCode()))
                {
                    // 5. 数据库查询账号
                    DBComponent dbComponent = session.Root().GetComponent<DBManagerComponent>().GetZoneDB(session.Zone());
                    List<Account> accountInfoList = null;
                    try 
                    {
                        accountInfoList = await dbComponent.Query<Account>(d => d.AccountName.Equals(request.AccountName));
                    }
                    catch (Exception e)
                    {
                        Log.Error($"数据库操作失败: {e}");
                    }

                    Account account = null;
                    if (accountInfoList != null && accountInfoList.Count > 0)
                    {
                        // 6. 账号存在，验证密码和黑名单
                        account = accountInfoList[0];
                        session.AddChild(account);

                        if (account.AccountType == (int)AccountType.BlackList)
                        {
                            response.Error = ErrorCode.ERR_AccountInBlackListError;
                            session.Disconnect().Coroutine();
                            account?.Dispose();
                            return;
                        }

                        if (!account.Password.Equals(request.Password))
                        {
                            response.Error = ErrorCode.ERR_PasswordFormError;
                            session.Disconnect().Coroutine();
                            account?.Dispose();
                            return;
                        }
                    }
                    else
                    {
                        // 7. 账号不存在，创建新账号
                        account = session.AddChild<Account>();
                        account.AccountName = request.AccountName.Trim();
                        account.Password = request.Password;
                        account.CreateTime = TimeInfo.Instance.ServerNow();
                        account.AccountType = (int)AccountType.General;
                        await dbComponent.Save<Account>(account);
                    }

                    // 8. 向LoginCenter请求登录验证
                    R2L_LoginAccountRequest r2LLoginAccountRequest = R2L_LoginAccountRequest.Create();
                    r2LLoginAccountRequest.AccountName = request.AccountName;

                    StartSceneConfig loginCenterConfig = StartSceneConfigCategory.Instance.LoginCenterConfig;
                    L2R_LoginAccountRequest loginAccountResponse = await session.Fiber().Root.GetComponent<MessageSender>()
                        .Call(loginCenterConfig.ActorId, r2LLoginAccountRequest) as L2R_LoginAccountRequest;

                    if (loginAccountResponse.Error != ErrorCode.ERR_Success)
                    {
                        response.Error = loginAccountResponse.Error;
                        session?.Disconnect().Coroutine();
                        account?.Dispose();
                        return;
                    }

                    // 9. 顶号处理 - 断开同账号的其他Session
                    Session otherSession = session.Root().GetComponent<AccountSessionComponent>().Get(request.AccountName);
                    otherSession?.Disconnect().Coroutine();

                    // 10. 生成Token
                    string Token = TimeInfo.Instance.ServerNow().ToString() + 
                                   RandomGenerator.RandomNumber(int.MinValue, int.MaxValue).ToString();
                    response.Token = Token;

                    // 11. 记录Session和Token
                    session.Root().GetComponent<AccountSessionComponent>().Add(request.AccountName, session);
                    session.AddComponent<AccountCheckOutTimeComponent, string>(request.AccountName);
                    
                    session.Root().GetComponent<TokenComponent>().Remove(request.AccountName);
                    session.Root().GetComponent<TokenComponent>().Add(request.AccountName, Token);

                    account?.Dispose();        
                }
            }
        }
    }
}
```

```csharp
namespace ET.Server
{
    [MessageSessionHandler(SceneType.Gate)]
    public class C2G_LoginGameGateHandler : MessageSessionHandler<C2G_LoginGameGate, G2C_LoginGameGate>
    {
        protected override async ETTask Run(Session session, C2G_LoginGameGate request, G2C_LoginGameGate response)
        {
            Scene root = session.Root();

            // 1. 检查重复请求
            if (session.GetComponent<SessionLockingComponent>() != null)
            {
                response.Error = ErrorCode.ERR_RequestRepeatedly;
                return;
            }

            // 2. 验证Gate Key
            string account = root.GetComponent<GateSessionKeyComponent>().Get(request.Key);
            if (account == null)
            {
                response.Error = ErrorCode.ERR_ConnectGateKeyError;
                response.Message = "Gate Key验证失败";
                session?.Disconnect().Coroutine();
                return;
            }
            
            // 3. 移除已使用的Key
            root.GetComponent<GateSessionKeyComponent>().Remove(request.Key);
            session.RemoveComponent<SessionAcceptTimeoutComponent>();

            CoroutineLockComponent coroutineLockComponent = root.GetComponent<CoroutineLockComponent>();
            long instanceId = session.InstanceId;

            using (session.AddComponent<SessionLockingComponent>())
            {
                using (await coroutineLockComponent.Wait(CoroutineLockType.LoginGate, request.AccountName.GetLongHashCode()))
                {
                    if (instanceId != session.InstanceId)
                    {
                        response.Error = ErrorCode.ERR_LoginGameGateError;
                        return;
                    }
                    
                    // 4. 通知LoginCenter记录本次登录
                    G2L_AddLoginRecord g2LAddLoginRecord = G2L_AddLoginRecord.Create();
                    g2LAddLoginRecord.AccountName = request.AccountName;
                    g2LAddLoginRecord.ServerId = root.Zone();

                    L2G_AddLoginRecord l2GALoginRecord = (L2G_AddLoginRecord)await root.GetComponent<MessageSender>()
                        .Call(StartSceneConfigCategory.Instance.LoginCenterConfig.ActorId, g2LAddLoginRecord);

                    if (l2GALoginRecord.Error != ErrorCode.ERR_Success)
                    {
                        response.Error = l2GALoginRecord.Error;
                        session?.Disconnect().Coroutine();
                        return;
                    }

                    // 5. 创建或获取Player实体
                    PlayerComponent playerComponent = root.GetComponent<PlayerComponent>();
                    Player player = playerComponent.GetByAccount(account);

                    if (player == null)
                    {
                        // 新玩家 - 创建Player实体
                        player = playerComponent.AddChildWithId<Player, string>(id: request.RoleId, account);
                        player.UnitId = request.RoleId;
                        
                        playerComponent.Add(player);

                        // 添加Session组件并注册Location
                        PlayerSessionComponent playerSessionComponent = player.AddComponent<PlayerSessionComponent>();
                        playerSessionComponent.AddComponent<MailBoxComponent, MailBoxType>(MailBoxType.GateSession);
                        await playerSessionComponent.AddLocation(LocationType.GateSession);

                        player.AddComponent<MailBoxComponent, MailBoxType>(MailBoxType.UnOrderedMessage);
                        await player.AddLocation(LocationType.Player);

                        session.AddComponent<SessionPlayerComponent>().Player = player;
                        playerSessionComponent.Session = session;

                        player.PlayerState = PlayerState.Gate;
                    }
                    else
                    {
                        // 老玩家 - 恢复Session（断线重连）
                        player.RemoveComponent<PlayerOfflineOutTimeComponent>();
                        session.AddComponent<SessionPlayerComponent>().Player = player;
                        player.GetComponent<PlayerSessionComponent>().Session = session;
                    }

                    response.PlayerId = player.Id;
                }
            }
        }
    }
```

## 9. 注意事项

### 9.1 安全性考虑

1. **密码存储**：当前实现为明文存储，生产环境建议使用加密存储（如BCrypt、SHA256+Salt）
2. **Token生成**：当前使用时间戳+随机数，可考虑使用JWT或更安全的Token生成方式
3. **账号验证**：正则表达式可根据需求调整

### 9.2 扩展建议

1. **第三方登录**：可扩展支持微信、QQ、Apple等第三方登录
2. **验证码**：可添加图形验证码或短信验证码
3. **登录日志**：可添加登录日志记录，用于安全审计
4. **设备绑定**：可添加设备ID绑定功能

### 9.3 协程锁说明

登录流程中使用了多个协程锁防止并发问题：

| 锁类型 | 锁Key | 使用场景 |
|--------|-------|---------|
| LoginAccount | AccountName.GetLongHashCode() | Realm账号登录、LoginCenter登录验证 |
| LoginGate | AccountName.GetLongHashCode() | Gate网关登录 |
